      navTabs.forEach(nav => nav.classList.remove('active'));


      document.getElementById(tabName).classList.add('active');

    function updateProgress() {

      const currentMiles = parseFloat(document.getElementById('currentMiles').value) || 0;


      if (currentMiles < 0 || currentMiles > 1400) {

        alert('Please enter a value between 0 and 1400 miles');

        return;

      }


      const totalDistance = 1400;

      const progressPercent = (currentMiles / totalDistance) * 100;

      const milesRemaining = totalDistance - currentMiles;


      const mpg = 10;

      const startingFuel = 60;

      let fuelRemaining = startingFuel - (currentMiles % 600) / mpg;

      if (fuelRemaining < 0) fuelRemaining = 0;


      let milesUntilFuel;

      if (currentMiles < 600) {

        milesUntilFuel = 600 - currentMiles;

      } else if (currentMiles < 1200) {

        milesUntilFuel = 1200 - currentMiles;

      } else {

        milesUntilFuel = 0;

      }


      const cities = [

        { miles: 70, name: "Jackson, MS" },

        { miles: 230, name: "Shreveport, LA" },

        { miles: 430, name: "Dallas, TX" },

        { miles: 600, name: "Amarillo, TX (FUEL STOP)" },

        { miles: 875, name: "Tucumcari, NM" },

        { miles: 1055, name: "Albuquerque, NM" },

        { miles: 1200, name: "Raton, NM (FUEL STOP)" },

        { miles: 1330, name: "Pueblo, CO" },

        { miles: 1400, name: "Peyton, CO (DESTINATION!)" }

      ];


      let nextCity = "Peyton, CO (DESTINATION!)";

      for (let city of cities) {

        if (currentMiles < city.miles) {

          nextCity = city.name;

          break;

        }

      }


      const avgGasPrice = 3.72;

      const fuelNeededToFinish = milesRemaining / mpg;

      const fuelCostRemaining = (fuelNeededToFinish * avgGasPrice).toFixed(2);


      document.getElementById('progressBar').style.width = progressPercent + '%';

      document.getElementById('progressBar').textContent = progressPercent.toFixed(1) + '%';

      document.getElementById('milesTraveled').textContent = currentMiles.toFixed(0);

      document.getElementById('milesRemaining').textContent = milesRemaining.toFixed(0);

      document.getElementById('fuelRemaining').textContent = fuelRemaining.toFixed(1);

      document.getElementById('milesUntilFuel').textContent =

        milesUntilFuel > 0 ? milesUntilFuel.toFixed(0) : 'No more fuel stops needed!';

      document.getElementById('nextCity').textContent = nextCity;

      document.getElementById('fuelCostRemaining').textContent = fuelCostRemaining;

    }


    function resetProgress() {

      document.getElementById('currentMiles').value = 0;

      updateProgress();

    }


    window.onload = function() {

      updateProgress();

    };

  </script>
